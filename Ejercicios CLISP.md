```

(+ (+ (+ 3 6) (* 8 3 5)) 6) + 3

(+ (+ (+ (+ 3 6) (* 8 3 5)) 6) 3) + (* 3 2)
(+ (+ (+ (+ (+ 3 6) (* 8 3 5)) 6) 3) (* 3 2))

(+ (+ (+ (+ (+ 3 6) (* 8 3 5)) 6) 3) (* 3 2)) - 4

(+ (+ (+ (+ (+ (+ (+ 3 6) (* 8 3 5)) 6) 3) (* 3 2)) (- 4)) (/ 10 2))

```
